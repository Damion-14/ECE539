def get_image_from_url(url):
    response = requests.get(url,headers={'User-Agent': 'Mozilla/5.0'})
    img = Image.open(BytesIO(response.content))
    img = img.resize((600, 500))
    img = ImageOps.grayscale(img)
    return img
    
img = get_image_from_url(url)
